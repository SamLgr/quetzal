//Returns True if successful.
+destroyStack(): Boolean {query}

//Checks if stack is empty.
//Returns True if successful.
+isEmpty(): Boolean {query}

//Returns True if successful.

//Returns True if successful.

//Places top in 'stackTop' and returns 'stackTop'.
//Returns True if successful.
+getTop(out stackTop:StackItemType): Boolean {query}
